        if (isNaN(data.age) || isNaN(data.studyhours) || isNaN(data.classattendance) || isNaN(data.sleephours)) {
            errorEl.textContent = "Пожалуйста, заполните все числовые поля (возраст, часы учёбы, посещаемость, сон).";
            errorEl.style.display = "block";
            return;
        }

        try {
            const res = await fetch(API_URL, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify(data)
            });

            if (!res.ok) {
                throw new Error("Ошибка ответа сервера");
            }

            const json = await res.json();
            const prediction = typeof json.prediction === "number" ? json.prediction : null;
            const comment = json.comment || "";

            if (prediction == null) {
                throw new Error("Некорректный формат ответа модели");
            }

            const rounded = Math.round(prediction * 10) / 10;
            predScoreEl.textContent = rounded.toString();
            predCommentEl.textContent = comment || "Прогноз основан на твоём балансе учёбы, посещаемости и сна по обучающей выборке.";

            resultBlock.style.display = "block";
        } catch (err) {
            console.error(err);
            errorEl.textContent = "Не удалось получить предсказание. Проверь backend‑эндпоинт или попробуй позже.";
            errorEl.style.display = "block";
        }
    });
